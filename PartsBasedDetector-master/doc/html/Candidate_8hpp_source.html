119  cv::Mat_<int> xpts(cv::Size(1,nparts));
120  cv::Mat_<int> ypts(cv::Size(1,nparts));
121  for (unsigned int n = 0; n < nparts; ++n) {

123  xpts(n) = centroid.x;

140  Rect3d boundingBox3D(const cv::Mat& im, const cv::Mat& depth) const {
141 
142  const unsigned int nparts = parts_.size();
143  const cv::Rect bounds = cv::Rect(0,0,0,0) + im.size();
144  const cv::Rect bb = this->boundingBox();
145  const cv::Rect bbn = this->boundingBoxNorm();
146 
147  cv::Size_<double> imsize = im.size();
148  cv::Size_<double> dsize = depth.size();
149  cv::Point_<double> s = cv::Point_<double>(dsize.width / imsize.width, dsize.height / imsize.height);
150 
151  cv::Mat_<float> points;
152  std::vector<cv::Rect> boxes;
153  for (unsigned int n = 0; n < nparts; ++n) {
154  // only keep the intersection of the part with the image frame
155  boxes.push_back(parts_[n] & bounds);
156  }
157  boxes.push_back(bbn & bounds);
158 
159  for (unsigned int n = 0; n < boxes.size(); ++n) {
160  // scale the part down to match the depth image size
161  cv::Rect& r = boxes[n];
162  r.x = r.x * s.x;
163  r.y = r.y * s.y;
164  r.width = r.width * s.x;
165  r.height = r.height * s.y;
166 
167  // add the valid points
168  cv::Mat_<float> part = depth(r);
169  if(part.empty())
170  continue;
171 
172  for (cv::MatIterator_<float> it = part.begin(); it != part.end(); ++it) {
173  if (*it != 0 && !std::isnan(*it)) points.push_back(*it);
174  }
175 
176  if(points.empty())
177  {
178  return Rect3d(std::numeric_limits<double>::quiet_NaN(), std::numeric_limits<double>::quiet_NaN(), std::numeric_limits<double>::quiet_NaN(),
179  0, 0, 0);
180  }
181 
182  }
183 
184  // sort the points
185  std::sort(points.begin(), points.end());
186  cv::resize(points, points, cv::Size(1, 400));
187 
188  // get the median of the points
189  const unsigned int M = points.rows;
190  const unsigned int midx = M/2;
191  float median = points[midx][0];
192 
193  // filter the points
194  cv::Mat_<float> g, dog, dpoints;
195  cv::Matx<float, 3, 1> diff(-1, 0, 1);
196  g= cv::getGaussianKernel(35, 4, CV_32F);
197  cv::filter2D(g, dog, -1, diff);
198  cv::filter2D(points, dpoints, -1, dog);
199 
200  // starting at the median point, walk up and down until a gradient threshold (0.1)
201  unsigned int dminidx = midx, dmaxidx = midx;
202  for (unsigned int m = midx; m < M; ++m) {
203  if (fabs(dpoints[m][0]) > 0.035) break;
204  dmaxidx = m;
205  }
206  for (int m = midx; m >= 0; --m) {
207  if (fabs(dpoints[m][0]) > 0.035) break;
208  dminidx = m;
209  }
210 
211  // construct the 3D bounding box
212  cv::Point3_<double> tl(bb.x, bb.y, points[dminidx][0]);
213  cv::Point3_<double> br(bb.br().x, bb.br().y, points[dmaxidx][0]);
214 
215  return Rect3d(tl, br);
216  }

277  static void nonMaximaSuppression(const cv::Mat& im, vectorCandidate& candidates, const float overlap=0.0f) {
278 
279  // create a scratch space that we can draw on
280  const unsigned int N = candidates.size();
281  const cv::Rect bounds = cv::Rect(0,0,0,0) + im.size();
282  cv::Mat scratch = cv::Mat::zeros(im.size(), CV_8U);
283 
284  // the current insertion position in the vector
285  unsigned int keep = 0;
286 
287  /* iterate through the boxes, checking:
288  * 1) has the area under the box been painted?
289  * 2) if so, is it under the threshold?
290  * 3) if so, keep this box and paint the area
291  * 4) repeat
292  */
293  for (unsigned int n = 0; n < N; ++n) {
294  cv::Rect box = candidates[n].boundingBox() & bounds;
295  cv::Scalar boxsum = sum(scratch(box));
296  if (boxsum[0] / box.area() > overlap) continue;
297  scratch(box) = 1;
298  candidates[keep] = candidates[n];
299  keep++;
300  }
301 
302  // simply delete the trailing end of the candidates
303  candidates.resize(keep);
304  }
