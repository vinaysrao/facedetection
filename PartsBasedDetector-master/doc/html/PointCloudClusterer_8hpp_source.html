52 template<typename PointType>
53 void PointCloudClusterer<PointType>::computeBoundingBoxes(
54  const std::vector<Candidate>& candidates, const cv::Mat& rgb,
55  const cv::Mat& depth, PointProjectFunc camera_model_projecter,
56  const typename PointCloud::ConstPtr cloud,
57  std::vector<Rect3d>& bounding_boxes,
58  std::vector<PointCloud>& parts_centers)
59 {
60  bounding_boxes.clear();
61  parts_centers.clear();
62 
63  if (candidates.size() == 0)
64  {
65  return;
66  }
67 
68  double ticks = (double) cv::getTickCount();
69 
70  bounding_boxes.resize(candidates.size(), Rect3d(0, 0, 0, 0, 0, 0));
71  parts_centers.resize(candidates.size());
72 
73  for (size_t i = 0; i < candidates.size(); ++i)
74  {
75  const Candidate& candidate = candidates[i];
76 
77  const Rect3d cube = candidate.boundingBox3D(rgb, depth);
78  cv::Point3d tl, br;
79 
80  if (isnan(cube.x) || isnan(cube.y) || isnan(cube.z) || isnan(cube.width)
81  || isnan(cube.height) || isnan(cube.depth))
82  {
83  std::cout << "Candidate " << i << " bounding box contains nans."
84  << std::endl;
85  //ROS_INFO("Candidate %zu bounding box contains nans.", i);
86  continue;
87  }
88 
89  std::cout << "Score: " << candidate.score() << " - Cube data: "
90  << cube.tl().x << " " << cube.tl().y << " " << cube.z << " "
91  << cube.br().x << " " << cube.br().y << " " << cube.depth
92  << std::endl;
93 // ROS_INFO("Score: %f - Cube data: %f %f %f %f %f %f", candidate.score(),
94 // cube.tl().x, cube.tl().y, cube.z, cube.br().x, cube.br().y,
95 // cube.depth);
96 
97  for (size_t j = 0; j < candidate.parts().size(); ++j)
98  {
99 // ROS_INFO("\tPart %zu: %d %d -> %d %d conf: %f", j,
100 // candidate.parts()[j].x, candidate.parts()[j].y,
101 // candidate.parts()[j].height, candidate.parts()[j].width,
102 // candidate.confidence()[j]);
103  cv::Rect part = candidate.parts()[j];
104 
105  //clamp the part to the image
106  part &= cv::Rect(cv::Point(0, 0), rgb.size());
107  cv::Point center(part.x + (part.width / 2),
108  part.y + (part.height / 2));
109 
110  double avg_depth = 0.0;
111  for (int row_it = part.x; row_it < part.x + part.height; ++row_it)
112  {
113  const float *row_ptr = depth.ptr<float>(row_it);
114  for (int col_it = part.y; col_it < part.y + part.width;
115  ++col_it)
116  {
117  avg_depth += row_ptr[col_it];
118  }
119  }
120 
121  if (part.width * part.height != 0)
122  {
123  avg_depth /= part.width * part.height;
124  }
125 
126  // To keep this independent from camera/point cloud resolution difference:
127  const cv::Point3d point = camera_model_projecter(center)
128  * avg_depth; // depth.at<float>(center);
129 // const PointType& out_point = cloud.at(center.x, center.y);
130 
131  PointType out_point;
132  out_point.x = point.x;
133  out_point.y = point.y;
134  out_point.z = point.z;
135 
136  if (isnan(out_point.x) || isnan(out_point.y) || isnan(out_point.z))
137  {
138  parts_centers[i].is_dense = false;
139  }
140 
141  parts_centers[i].push_back(out_point);
142  }
143  tl = camera_model_projecter(cv::Point2d(cube.tl().x, cube.tl().y))
144  * cube.z;
145  br = camera_model_projecter(cv::Point2d(cube.br().x, cube.br().y))
146  * (cube.z + cube.depth);
147 
148  bounding_boxes[i] = Rect3d(tl, br);
149  }
150 
151  std::cout << "Bounding boxes computation time: "
152  << ((double) cv::getTickCount() - ticks) / cv::getTickFrequency()
153  << std::endl;
154 }

156 template<typename PointType>
157 void PointCloudClusterer<PointType>::clusterObjects(
158  const PointCloudConstPtr cloud,
159  const std::vector<Rect3d>& bounding_boxes,
160  std::vector<PointCloud>& object_clusters,
161  std::vector<PointType>& object_centers)
162 {
163  object_clusters.clear();
164  object_centers.clear();
165 
166  if (bounding_boxes.size() == 0)
167  {
168  return;
169  }
170 
171  double ticks = (double) cv::getTickCount();
172 
173  {
174  PointType nan_point;
175  nan_point.x = std::numeric_limits<float>::quiet_NaN();
176  nan_point.y = std::numeric_limits<float>::quiet_NaN();
177  nan_point.z = std::numeric_limits<float>::quiet_NaN();
178 
179  object_clusters.resize(bounding_boxes.size());
180  object_centers.resize(bounding_boxes.size(), nan_point);
181 
182  }
183 
184  std::vector<pcl::PointIndices::Ptr> boxed_indices(bounding_boxes.size());
185  //boost::shared_ptr<std::vector<int> > final_indices(new std::vector<int>());
186 
187  // crop around each bounding box
188  pcl::CropBox<PointType> crop_box;
189  crop_box.setInputCloud(cloud);
190 
191  for (size_t i = 0; i < bounding_boxes.size(); ++i)
192  {
193  pcl::PointIndices::Ptr indices(new pcl::PointIndices());
194 
195  // expand a bit the box
196  Rect3d bbox = bounding_boxes[i];
197 
198  if(bbox.volume() >= 1E-6) // 1 cubic centimeter
199  {
200  bbox -= cv::Point3d(bbox.width * 0.1, bbox.height * 0.1,
201  bbox.depth * 0.1);
202  bbox.width *= 1.2;
203  bbox.height *= 1.2;
204  bbox.depth *= 1.2;
205 
206  crop_box.setMin(
207  Eigen::Vector4f(bbox.tl().x, bbox.tl().y, bbox.tl().z, 0));
208  crop_box.setMax(
209  Eigen::Vector4f(bbox.br().x, bbox.br().y, bbox.br().z, 0));
210 
211  crop_box.filter(indices->indices);
212 // std::cout << "Cropped obj " << i << ", remaining indices: " << indices->indices.size() << std::endl;
213  }
214  else
215  {
216 // std::cout << "Bounding box for obj " << i << " empty." << std::endl;
217  }
218  // add the points belonging to the bbox
219  boxed_indices[i] = indices;
220  }
221 
222  std::cout << "Cropping ok..." << std::endl;
223 
224  // Extract different clusters to remove parts of other objects appearing into the bounding boxes
225  pcl::EuclideanClusterExtraction<PointType> boxes_clusterer;
226  boxes_clusterer.setInputCloud(cloud);
227  boxes_clusterer.setClusterTolerance(0.010);
228 
229  pcl::ExtractIndices<PointType> extract_indices;
230  extract_indices.setInputCloud(cloud);
231 
232  for (int i = 0; i < bounding_boxes.size(); ++i)
233  {
234  std::vector<pcl::PointIndices> clusters;
235 
236  if (!boxed_indices[i] || boxed_indices[i]->indices.size() == 0)
237  {
238 // ROS_INFO("Empty boxes indices %d", i);
239  continue;
240  }
241 
242  std::cout << "Start clustering " << i << std::endl;
243  // keep the biggest one (should be the object)
244  boxes_clusterer.setIndices(boxed_indices[i]);
245  boxes_clusterer.extract(clusters);
246 // std::cout << "Clusterer extraction for object " << i << " succeeded with " << clusters.size() << " results" << std::endl;
247 
248  // ROS_INFO("Extracted %zu clusters", clusters.size());
249 
250  // Keep only the maximum cluster
251  if (clusters.size() > 0)
252  {
253  int max_idx = 0;
254  for (int j = 1; j < clusters.size(); ++j)
255  {
256  if (clusters[j].indices.size()
257  > clusters[max_idx].indices.size())
258  {
259  max_idx = j;
260  }
261  }
262 
263  //compute centroid
264  Eigen::Vector4f centroid;
265 
266  int point_count = 0;
267  if ((point_count = pcl::compute3DCentroid(*cloud,
268  clusters[max_idx].indices, centroid)) == 0)
269  {
270 // ROS_WARN("Centroid not found...");
271  continue;
272  }
273 
274  PointType center_point;
275  center_point.x = centroid.x();
276  center_point.y = centroid.y();
277  center_point.z = centroid.z();
278 
279  // add pose
280  object_centers[i] = center_point;
281 
282  // extract cluster from main cloud
283  extract_indices.setIndices(
284  boost::make_shared<pcl::PointIndices>(clusters[max_idx]));
285  extract_indices.filter(object_clusters[i]);
286  }
287  }
288 
289  std::cout << "Clustering time: "
290  << ((double) cv::getTickCount() - ticks) / cv::getTickFrequency()
291  << std::endl;
292 }

294 template<typename PointType>
295 void PointCloudClusterer<PointType>::organizedMultiplaneSegmentation(
296  const PointCloudConstPtr& cloud, PointCloud& cloud_no_plane)
297 {
298  pcl::PointCloud<pcl::Normal>::Ptr normals(
299  new pcl::PointCloud<pcl::Normal>());
300  pcl::IntegralImageNormalEstimation<PointType, pcl::Normal> ne;
301  ne.setInputCloud(cloud);
302  ne.compute(*normals);
303 
304  pcl::OrganizedMultiPlaneSegmentation<PointType, pcl::Normal, pcl::Label> plane_segmentation;
305  plane_segmentation.setInputCloud(cloud);
306  plane_segmentation.setInputNormals(normals);
307  plane_segmentation.setDistanceThreshold(0.02);
308  //plane_segmentation.setAngularThreshold(pcl::deg2rad(5.0));
309  plane_segmentation.setMaximumCurvature(0.001);
310  plane_segmentation.setProjectPoints(true);
311 
312  std::vector<pcl::PlanarRegion<PointType>,
313  Eigen::aligned_allocator<pcl::PlanarRegion<PointType> > > regions;
314  std::vector<pcl::ModelCoefficients> model_coefficients;
315  std::vector<pcl::PointIndices> inlier_indices;
316  pcl::PointCloud<pcl::Label>::Ptr labels(new pcl::PointCloud<pcl::Label>());
317  std::vector<pcl::PointIndices> label_indices;
318  std::vector<pcl::PointIndices> boundary_indices;
319 
320  plane_segmentation.segmentAndRefine(regions, model_coefficients,
321  inlier_indices, labels, label_indices, boundary_indices);
322 
323  boost::shared_ptr<std::vector<int> > inliers(new std::vector<int>());
324  for (int i = 0; i < inlier_indices.size(); ++i)
325  {
326  inliers->insert(inliers->end(), inlier_indices[i].indices.begin(),
327  inlier_indices[i].indices.end());
328  }
329 
330  pcl::ExtractIndices<PointType> extract_indices;
331  extract_indices.setInputCloud(cloud);
332  extract_indices.setIndices(inliers);
333  extract_indices.setNegative(true);
334  extract_indices.filter(cloud_no_plane);
335 }
